tomcat direct war file deployment:
---------------------------------
FROM tomcat:8.0.20-jre8              ----->jre must for any version
MAINTAINER roja <krojakumari99@gmail.com>
WORKDIR /usr/local/tomcat/webapps/
RUN sed -i 's/8080/9090/' /usr/local/tomcat/conf/server.xml  --->for chenging port no. of tomcat
EXPOSE 9090            --->JUST for identify the port this is for changing port 
EXPOSE 8080            ---->normal without changing port no.
COPY money-transfer.war /usr/local/tomcat/webapps/money-transfer.war
 
after docker file
$docker build -t tomcat .
$docker images
$docker run -itd -p 8080:8080 --name money-transerapp imageid
